import streamlit as st
import requests
import pandas as pd

st.set_page_config(page_title="Analista Pro Fut", layout="wide")
st.title("⚽ Dashboard de Análise Estatística")

api_key = 'c9f55dc0e91ac61caf1e05371daa9a8c'
headers = {'x-rapidapi-key': api_key}

# Lista de IDs para monitorar (Brasileirão 2024 para ter dados)
times_ids = {"Flamengo": 127, "Palmeiras": 121, "Botafogo": 120, "São Paulo": 126, "Bahia": 118}

if st.button('🚀 Gerar Relatório de Oportunidades'):
    dados_finais = []
    
    for nome, tid in times_ids.items():
        url = f"https://v3.football.api-sports.io/teams/statistics?league=71&season=2024&team={tid}"
        res = requests.get(url, headers=headers).json()['response']
        
        # Lógica de Ambas Marcam (Gols feitos e sofridos > 0)
        gols_feitos = res['goals']['for']['average']['total']
        gols_sofridos = res['goals']['against']['average']['total']
        btts_vibe = "✅ ALTA" if float(gols_feitos) > 1.1 and float(gols_sofridos) > 1.0 else "❌ BAIXA"
        
        dados_finais.append({
            "Time": nome,
            "Escanteios (Média)": res['corners']['avg']['total'],
            "Amarelos (Total)": res['cards']['yellow']['total'],
            "Tendência AMBAS": btts_vibe,
            "Vitórias": res['fixtures']['wins']['total']
        })

    df = pd.DataFrame(dados_finais)
    
    # Criando colunas no layout do app
    col1, col2 = st.columns(2)
    with col1:
        st.subheader("🔥 Melhores para Escanteios")
        st.table(df[['Time', 'Escanteios (Média)']].sort_values(by='Escanteios (Média)', ascending=False))
    
    with col2:
        st.subheader("🟨 Tendência de Cartões")
        st.table(df[['Time', 'Amarelos (Total)']].sort_values(by='Amarelos (Total)', ascending=False))

    st.subheader("📊 Resumo Ambas Marcam")
    st.dataframe(df[['Time', 'Tendência AMBAS', 'Vitórias']].style.applymap(
        lambda x: 'color: green' if x == "✅ ALTA" else 'color: red', subset=['Tendência AMBAS']
    ))
